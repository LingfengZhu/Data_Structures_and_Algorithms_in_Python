void CreateLinkListTail(LinkList *L,int n){
  LinkList p,r;
  srand(time(0));
  *L=(LinkList)malloc(sizeof(LNode));
  r=*L;
  for(i=0;i<n;i++){
    p=(Lnode *)malloc(sizeof(LNode));
    p->data=rand()%100+1;
    r->next=p;
    r=p;
  }
  r->next=NULL;
}
