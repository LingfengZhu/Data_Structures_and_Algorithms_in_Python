Status SqListInsert(SqList *L,int i,ElemType e){
  int j;
  if (L->length>=MAX_SIZE) { //`线性表已满`
    printf("`线性表溢出！`\n"); return ERROR;
  }
  if (i<0||i>L->length-1)   //`i不在范围内`
    return ERROR;
  for (j=L->length-1; j>=i-1; j--)
    L->data[j+1]=L->data[j];
  L->data[i-1]=e;
  L->length++;
  return OK;
}
